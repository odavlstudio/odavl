/**
 * ODAVL Recipe Engine - Core Execution System
 * 
 * Provides the main recipe discovery, selection, and execution engine.
 * This replaces the hardcoded decision logic with a flexible plugin system.
 */
 
import * as fs from 'fs';
import * as path from 'path';
import { spawnSync } from 'child_process';
import { Recipe, RecipePlugin, RecipeContext, RecipeResult, TrustRecord } from './types.js';
 
export class RecipeEngine {
  private plugins: Map<string, RecipePlugin> = new Map();
  private context: RecipeContext;
  
  constructor(context: RecipeContext) {
    this.context = context;
    this.loadPlugins();
  }
  
  /**
   * Dynamically discovers and loads recipe plugins from .odavl/recipes directory
   */
  private loadPlugins(): void {
    const recipesDir = path.join(this.context.odavlDir, 'recipes');
    
    // Load recipes from .odavl/recipes directory
    if (fs.existsSync(recipesDir)) {
      this.loadRecipesFromDirectory(recipesDir, false);
    }
  }
  
  private loadRecipesFromDirectory(directory: string, _isBuiltin: boolean): void {
    if (!fs.existsSync(directory)) return;
    
    for (const filename of fs.readdirSync(directory)) {
      if (!filename.endsWith('.json') && !filename.endsWith('.js') && !filename.endsWith('.ts')) {
        continue;
      }
      
      try {
        const filepath = path.join(directory, filename);
        let plugin: RecipePlugin;
        
        if (filename.endsWith('.json')) {
          // JSON-based recipe (legacy compatibility)
          const recipeData = JSON.parse(fs.readFileSync(filepath, 'utf8'));
          plugin = this.createJsonRecipePlugin(recipeData);
        } else {
          // JavaScript/TypeScript module - skip for now as we need proper ES module loading
          console.warn(`Skipping JS/TS recipe ${filename} - ES module loading not yet implemented`);
          continue;
        }
        
        if (this.validatePlugin(plugin)) {
          this.plugins.set(plugin.recipe.id, plugin);
        }
      } catch (error) {
        console.warn(`Failed to load recipe from ${filename}: ${error instanceof Error ? error.message : String(error)}`);
      }
    }
  }
  
  /**
   * Creates a plugin wrapper for legacy JSON-based recipes
   */
  private createJsonRecipePlugin(recipeData: any): RecipePlugin { // eslint-disable-line @typescript-eslint/no-explicit-any
    // Convert legacy JSON format to new Recipe interface
    const recipe: Recipe = {
      id: recipeData.id,
      name: recipeData.name || recipeData.id,
      description: recipeData.description || '',
      version: recipeData.version || '1.0.0',
      priority: recipeData.priority || 50,
      trust: recipeData.trust,
      triggers: {
        eslintWarnings: recipeData.id === 'remove-unused' ? { min: 1 } : undefined,
        typeErrors: recipeData.id === 'type-fixes' ? { min: 1 } : undefined,
      },
      safety: {
        maxFilesTouched: 10,
        maxLinesChanged: 100,
        protectedPaths: ['node_modules/**', '.git/**'],
        requiresBackup: true
      },
      actions: [{
        type: 'command',
        command: this.getLegacyCommand(recipeData.id)
      }],
      tags: recipeData.tags || [recipeData.id],
      riskLevel: 'low'
    };
    
    return {
      recipe,
      canExecute: (context: RecipeContext) => this.checkTriggers(recipe, context),
      execute: async (context: RecipeContext) => this.executeLegacyRecipe(recipe, context),
    };
  }
  
  private getLegacyCommand(recipeId: string): string {
    switch (recipeId) {
      case 'remove-unused':
      case 'esm-hygiene':
      case 'format-consistency':
        return 'pnpm -s exec eslint . --fix';
      default:
        return 'echo "No operation defined"';
    }
  }
  
  private checkTriggers(recipe: Recipe, context: RecipeContext): boolean {
    const { triggers } = recipe;
    const { metrics } = context;
    
    // Check ESLint warnings trigger
    if (triggers.eslintWarnings) {
      const { min, max } = triggers.eslintWarnings;
      if (min !== undefined && metrics.eslintWarnings < min) return false;
      if (max !== undefined && metrics.eslintWarnings > max) return false;
    }
    
    // Check type errors trigger
    if (triggers.typeErrors) {
      const { min, max } = triggers.typeErrors;
      if (min !== undefined && metrics.typeErrors < min) return false;
      if (max !== undefined && metrics.typeErrors > max) return false;
    }
    
    // Check custom condition
    if (triggers.customCondition) {
      return triggers.customCondition(context);
    }
    
    return true;
  }
  
  private async executeLegacyRecipe(recipe: Recipe, context: RecipeContext): Promise<RecipeResult> {
    const action = recipe.actions[0];
    if (action.type !== 'command' || !action.command) {
      return { success: false, message: 'Invalid legacy recipe action' };
    }
    
    try {
      // Use cross-platform command execution
      const isWindows = process.platform === 'win32';
      const shell = isWindows ? 'cmd' : 'sh';
      const shellFlag = isWindows ? '/c' : '-c';
      
      const result = spawnSync(shell, [shellFlag, action.command], {
        cwd: context.projectRoot,
        stdio: 'pipe'
      });
      
      return {
        success: result.status === 0,
        message: result.status === 0 ? 'Command executed successfully' : `Command failed: ${result.stderr?.toString()}`,
        filesModified: [] // Legacy recipes don't track individual files
      };
    } catch (error) {
      return {
        success: false,
        message: `Execution error: ${error instanceof Error ? error.message : String(error)}`
      };
    }
  }
  
  private validatePlugin(plugin: RecipePlugin): boolean {
    return !!(
      plugin &&
      plugin.recipe &&
      plugin.recipe.id &&
      plugin.recipe.name &&
      typeof plugin.canExecute === 'function' &&
      typeof plugin.execute === 'function'
    );
  }
  
  /**
   * Decides which recipe to execute based on current metrics and trust scores
   */
  public selectRecipe(): RecipePlugin | null {
    const eligiblePlugins = Array.from(this.plugins.values())
      .filter(plugin => plugin.canExecute(this.context));
    
    if (eligiblePlugins.length === 0) {
      return null;
    }
    
    // Load trust scores
    const trustScores = this.loadTrustScores();
    
    // Sort by trust score (descending) then by priority (descending)
    const sortedPlugins = eligiblePlugins.sort((a, b) => {
      const aTrust = trustScores.get(a.recipe.id)?.trust ?? a.recipe.trust ?? 0.5;
      const bTrust = trustScores.get(b.recipe.id)?.trust ?? b.recipe.trust ?? 0.5;
      
      if (aTrust !== bTrust) {
        return bTrust - aTrust; // Higher trust first
      }
      
      return b.recipe.priority - a.recipe.priority; // Higher priority first
    });
    
    return sortedPlugins[0];
  }
  
  /**
   * Executes a recipe plugin with proper error handling and trust tracking
   */
  public async executeRecipe(plugin: RecipePlugin): Promise<RecipeResult> {
    const startTime = Date.now();
    
    try {
      // Create backup if required
      if (plugin.recipe.safety.requiresBackup) {
        await this.createBackup(plugin.recipe.id);
      }
      
      // Execute the recipe
      const result = await plugin.execute(this.context);
      
      // Update trust score based on result
      const duration = Date.now() - startTime;
      this.updateTrustScore(plugin.recipe.id, result.success, duration);
      
      return result;
    } catch (error) {
      // Update trust score for failure
      this.updateTrustScore(plugin.recipe.id, false, Date.now() - startTime);
      
      return {
        success: false,
        message: `Recipe execution failed: ${error instanceof Error ? error.message : String(error)}`
      };
    }
  }
  
  private loadTrustScores(): Map<string, TrustRecord> {
    const trustPath = path.join(this.context.odavlDir, 'recipes-trust.json');
    const trustMap = new Map<string, TrustRecord>();
    
    if (fs.existsSync(trustPath)) {
      try {
        const trustData: TrustRecord[] = JSON.parse(fs.readFileSync(trustPath, 'utf8'));
        for (const record of trustData) {
          trustMap.set(record.id, record);
        }
      } catch (error) {
        console.warn(`Failed to load trust scores: ${error instanceof Error ? error.message : String(error)}`);
      }
    }
    
    return trustMap;
  }
  
  private updateTrustScore(recipeId: string, success: boolean, duration: number): void {
    const trustPath = path.join(this.context.odavlDir, 'recipes-trust.json');
    let trustRecords: TrustRecord[] = [];
    
    if (fs.existsSync(trustPath)) {
      try {
        trustRecords = JSON.parse(fs.readFileSync(trustPath, 'utf8'));
      } catch (error) {
        console.warn(`Failed to load existing trust scores: ${error instanceof Error ? error.message : String(error)}`);
      }
    }
    
    let record = trustRecords.find(r => r.id === recipeId);
    if (!record) {
      record = {
        id: recipeId,
        runs: 0,
        success: 0,
        trust: 0.5,
        lastRun: new Date().toISOString(),
        averageDuration: 0
      };
      trustRecords.push(record);
    }
    
    // Update statistics
    record.runs++;
    if (success) record.success++;
    record.trust = Math.max(0.1, Math.min(1.0, record.success / record.runs));
    record.lastRun = new Date().toISOString();
    record.averageDuration = record.averageDuration 
      ? (record.averageDuration + duration) / 2 
      : duration;
    
    // Save updated trust scores
    try {
      fs.writeFileSync(trustPath, JSON.stringify(trustRecords, null, 2));
    } catch (error) {
      console.warn(`Failed to save trust scores: ${error instanceof Error ? error.message : String(error)}`);
    }
  }
  
  private async createBackup(recipeId: string): Promise<void> {
    // This would integrate with the existing undo snapshot system
    // For now, we'll just log that a backup should be created
    console.log(`Creating backup for recipe execution: ${recipeId}`);
  }
  
  /**
   * Lists all available recipes with their current trust scores
   */
  public listRecipes(): Array<{ plugin: RecipePlugin; trustScore: number }> {
    const trustScores = this.loadTrustScores();
    
    return Array.from(this.plugins.values()).map(plugin => ({
      plugin,
      trustScore: trustScores.get(plugin.recipe.id)?.trust ?? plugin.recipe.trust ?? 0.5
    }));
  }
}
