/**
 * Enhanced logging function with consistent formatting and colors
 */
export function log(message: string, type: 'info' | 'success' | 'warning' | 'error' | 'phase' = 'info'): void {
  const timestamp = colors.dim(new Date().toLocaleTimeString());
  let prefix = '';
  let coloredMessage = message;
  
  switch (type) {
    case 'success':
      prefix = colors.success(`${icons.success} `);
      coloredMessage = colors.success(message);
      break;
    case 'error':
      prefix = colors.error(`${icons.error} `);
      coloredMessage = colors.error(message);
      break;
    case 'warning':
      prefix = colors.warning(`${icons.warning} `);
      coloredMessage = colors.warning(message);
      break;
    case 'phase':
      prefix = colors.phase(`${icons.gear} `);
      coloredMessage = colors.phase(message);
      break;
    default:
      prefix = colors.info(`${icons.info} `);
      coloredMessage = colors.info(message);
  }
  
  console.log(`${timestamp} ${prefix}${coloredMessage}`);
}
 
/**
 * Enhanced phase logging with consistent ODAVL branding
 */
export function logPhase(phase: string, message: string, status: 'info' | 'success' | 'error' = 'info'): void {
  const phaseFormatted = colors.highlight(` ${phase} `);
  let statusIcon = icons.info;
  let messageColor = colors.info;
  
  switch (status) {
    case 'success':
      statusIcon = icons.success;
      messageColor = colors.success;
      break;
    case 'error':
      statusIcon = icons.error;
      messageColor = colors.error;
      break;
  }
  
  console.log(`${phaseFormatted} ${statusIcon} ${messageColor(message)}`);
}

/**
 * Progress spinner with customizable messages and phases
 */
export class ProgressSpinner {
  private spinner: Ora;
  private startTime: number;
  
  constructor(initialMessage: string, phase?: string) {
    const message = phase ? `[${colors.phase(phase)}] ${initialMessage}` : initialMessage;
    this.spinner = ora({
      text: message,
      spinner: 'dots',
      color: 'cyan'
    });
    this.startTime = Date.now();
  }
  
  start(): this {
    this.spinner.start();
    return this;
  }
  
  updateText(message: string, phase?: string): this {
    const text = phase ? `[${colors.phase(phase)}] ${message}` : message;
    this.spinner.text = text;
    return this;
  }
  
  succeed(message?: string): this {
    const elapsed = Date.now() - this.startTime;
    const elapsedText = colors.dim(`(${elapsed}ms)`);
    const finalMessage = message ? `${message} ${elapsedText}` : `${this.spinner.text} ${elapsedText}`;
    this.spinner.succeed(finalMessage);
    return this;
  }
  
  fail(message?: string): this {
    const elapsed = Date.now() - this.startTime;
    const elapsedText = colors.dim(`(${elapsed}ms)`);
    const finalMessage = message ? `${message} ${elapsedText}` : `${this.spinner.text} ${elapsedText}`;
    this.spinner.fail(finalMessage);
    return this;
  }
  
  warn(message?: string): this {
    const elapsed = Date.now() - this.startTime;
    const elapsedText = colors.dim(`(${elapsed}ms)`);
    const finalMessage = message ? `${message} ${elapsedText}` : `${this.spinner.text} ${elapsedText}`;
    this.spinner.warn(finalMessage);
    return this;
  }
  
  stop(): this {
    this.spinner.stop();
    return this;
  }
}
 
/**
 * Interactive prompts for user input
 */
export class InteractivePrompts {
  /**
   * Confirm an action with the user
   */
  static async confirm(message: string, defaultValue = false): Promise<boolean> {
    const answer = await inquirer.prompt([
      {
        type: 'confirm',
        name: 'confirmed',
        message: colors.info(message),
        default: defaultValue
      }
    ]);
    return answer.confirmed;
  }
  
  /**
   * Select from a list of options
   */
  static async select<T>(message: string, choices: Array<{ name: string; value: T }>): Promise<T> {
    const answer = await inquirer.prompt([
      {
        type: 'list',
        name: 'selected',
        message: colors.info(message),
        choices: choices.map(choice => ({
          name: choice.name,
          value: choice.value
        }))
      }
    ]);
    return answer.selected;
  }
  
  /**
   * Multiple selection from a list
   */
  static async multiSelect<T>(message: string, choices: Array<{ name: string; value: T }>): Promise<T[]> {
    const answer = await inquirer.prompt([
      {
        type: 'checkbox',
        name: 'selected',
        message: colors.info(message),
        choices: choices.map(choice => ({
          name: choice.name,
          value: choice.value
        }))
      }
    ]);
    return answer.selected;
  }
  
  /**
   * Text input with validation
   */
  static async input(message: string, validate?: (input: string) => boolean | string): Promise<string> {
    const answer = await inquirer.prompt([
      {
        type: 'input',
        name: 'input',
        message: colors.info(message),
        validate: validate
      }
    ]);
    return answer.input;
  }
  
  /**
   * Password input (hidden)
   */
  static async password(message: string): Promise<string> {
    const answer = await inquirer.prompt([
      {
        type: 'password',
        name: 'password',
        message: colors.info(message)
      }
    ]);
    return answer.password;
  }
}
 
/**
 * Format metrics display with consistent styling
 */
export function formatMetrics(metrics: { eslintWarnings: number; typeErrors: number; timestamp?: string }): string {
  const warnings = metrics.eslintWarnings > 0 
    ? colors.warning(`${metrics.eslintWarnings} warnings`)
    : colors.success('0 warnings');
  
  const errors = metrics.typeErrors > 0 
    ? colors.error(`${metrics.typeErrors} errors`)
    : colors.success('0 errors');
  
  const timestamp = metrics.timestamp 
    ? colors.dim(`(${new Date(metrics.timestamp).toLocaleTimeString()})`)
    : '';
  
  return `${warnings}, ${errors} ${timestamp}`;
}
 
/**
 * Format deltas with color coding for improvements/regressions
 */
export function formatDeltas(deltas: { eslint: number; types: number }): string {
  const eslintDelta = deltas.eslint === 0 
    ? colors.muted('¬±0')
    : deltas.eslint > 0 
      ? colors.error(`+${deltas.eslint}`)
      : colors.success(`${deltas.eslint}`);
  
  const typesDelta = deltas.types === 0 
    ? colors.muted('¬±0')
    : deltas.types > 0 
      ? colors.error(`+${deltas.types}`)
      : colors.success(`${deltas.types}`);
  
  return `ESLint: ${eslintDelta}, TypeScript: ${typesDelta}`;
}

/**
 * Terminal detection and capability checking
 */
export const terminal = {
  /**
   * Check if terminal supports colors
   */
  supportsColor(): boolean {
    return process.stdout.isTTY && !process.env.NO_COLOR;
  },
  
  /**
   * Check if terminal supports Unicode characters
   */
  supportsUnicode(): boolean {
    return process.platform !== 'win32' || process.env.WT_SESSION !== undefined;
  },
  
  /**
   * Get terminal width for responsive formatting
   */
  getWidth(): number {
    return process.stdout.columns || 80;
  }
};
